reprodataFilled <- reprodata
numRow <- nrow(reprodataFilled)
j =1

  if(is.na(reprodataFilled$steps[i])){
    reprodataFilled$steps[i] = reprodataMean$steps[j]
     numberOfReplacedNa=numberOfReplacedNa+1
  }
  j=j+1
  if(j==289){
    j=1
  }
}
